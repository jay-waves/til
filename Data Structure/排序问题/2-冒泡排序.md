## 原理：
- 每趟内层遍历，虽然单次比较的只是相邻两个元素，但是总能将最大（最小）的移动到最外围。
	> 因为遇到最大（最小）后，就始终有一个比较值是最大的那个
```c
#include <stdio.h>

void bubbleSort(int a[],int n);
void bubbleSort2(int a[],int n);

int main(){
	
	return 0;
}

void bubbleSort(int a[],int n)
{
	int hold;
	for(int i=0;i<n-1;i++){
		for(int j=0;j<n-1-i;j++){
			if(a[j]>a[j+1])
			{
				hold=a[j];
				a[j]=a[j+1];
				a[j+1]=hold;
			}
		}
	}
	return ;
}

void bubbleSort2(int a[],int n)
{
	int hold;
	for(int i=0;i<n-1;i++){
		int flag=0;
		for(int j=0;j<n-1-i;j++){
			if(a[j]>a[j+1])
			{
				hold=a[j];
				a[j]=a[j+1];
				a[j+1]=hold;
				flag=1;
			}
		}
		if(flag==0){//j循环后有交换则已，没有交换就推出 
			break;
		}
	}
}
```
